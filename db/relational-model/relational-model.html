        TIPO_USUARIO {
            int id_tipo_usuario PK
            string nombre_rol "Admin, Empleado, Cliente"
        }
        
        DOMICILIO {
            int id_domicilio PK
            string calle
            string numero_exterior
            string colonia
            string codigo_postal
            string ciudad
            string estado
        }
        
        USUARIOS {
            int id_usuario PK
            int id_domicilio FK
            int id_tipo_usuario FK
            string nombre
            string apellidos
            string correo UK
            string celular UK
            string password_hash
        }

        PERFIL_EMPLEADO {
            int id_usuario PK, FK
            string habilidad "Chofer, Cargador, Ambos"
            string numero_licencia
            string estado "Activo, Vacaciones, Baja"
        }
        
        CAMIONETAS {
            int id_camioneta PK
            string marca
            string modelo
            string placa UK
            decimal capacidad_m3
            string estado "Disponible, Mantenimiento, Baja"
        }
        
        VIAJES {
            int id_viaje PK
            int id_origen FK
            int id_destino FK
            datetime fecha_hora_inicio
            int duracion_estimada_min
            decimal distancia_km
            boolean anticipo_pagado
            string estado "Cotizado, Confirmado, En curso, Terminado, Cancelado"
            decimal costo_total
            string descripcion_articulos
        }

        VIAJE_CAMIONETAS {
            int id_viaje PK, FK
            int id_camioneta PK, FK
        }

        VIAJE_EMPLEADOS {
            int id_viaje PK, FK
            int id_empleado PK, FK
            string rol_en_este_viaje "Chofer, Auxiliar"
        }
        
        PAGOS {
            int id_pago PK
            int id_viaje FK
            decimal monto
            string metodo_pago "Efectivo, Transferencia"
            string referencia_bancaria "Null si es efectivo"
            datetime fecha_pago
            string estatus_pago "Pendiente, Liquidado"
        }

        CONFIG_COTIZADOR {
            int id_config PK
            decimal precio_km
            decimal precio_piso_base
            decimal sueldo_base_empleado
            datetime ultima_actualizacion
        }
        
        BITACORA {
            int id_bitacora PK
            int id_usuario FK
            string accion
            string tabla_afectada
            datetime fecha_hora
        }
